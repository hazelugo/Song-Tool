Create `middleware.ts` at the project root (per D-05, Next.js convention).

The middleware must:

1. **Import** `createServerClient` from `@supabase/ssr` and `NextResponse` from `next/server` and `type NextRequest` from `next/server`.

2. **Define protected route prefixes** as a constant array (per D-06):
   ```typescript
   const PROTECTED_PREFIXES = ["/songs", "/playlists", "/discovery"];
   ```

3. **Define the `middleware` function** (default export not needed — named export `middleware` is the Next.js convention):
   ```typescript
   export async function middleware(request: NextRequest) {
     const { pathname } = request.nextUrl;

     // Only check protected routes (per D-07: everything else passes through)
     const isProtected = PROTECTED_PREFIXES.some(
       (prefix) => pathname === prefix || pathname.startsWith(prefix + "/")
     );
     if (!isProtected) {
       return NextResponse.next();
     }

     // Create Supabase client using the middleware cookie pattern (per D-09)
     // In middleware, cookies() is not available — use request/response cookies directly
     let response = NextResponse.next({ request: { headers: request.headers } });

     const supabase = createServerClient(
       process.env.NEXT_PUBLIC_SUPABASE_URL!,
       process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
       {
         cookies: {
           getAll() {
             return request.cookies.getAll();
           },
           setAll(cookiesToSet) {
             cookiesToSet.forEach(({ name, value, options }) => {
               request.cookies.set(name, value);
               response = NextResponse.next({ request: { headers: request.headers } });
               response.cookies.set(name, value, options);
             });
           },
         },
       },
     );

     // Refresh session — IMPORTANT: must call getUser() not getSession()
     // getUser() validates the JWT with Supabase auth server
     const { data: { user } } = await supabase.auth.getUser();

     if (!user) {
       // Redirect to /login (per D-08)
       const loginUrl = request.nextUrl.clone();
       loginUrl.pathname = "/login";
       loginUrl.searchParams.delete("seedId"); // clean up any params
       return NextResponse.redirect(loginUrl);
     }

     return response;
   }
   ```

4. **Export matcher config** to limit middleware invocation to relevant paths (per D-07 — skip /_next, /api, static assets):
   ```typescript
   export const config = {
     matcher: [
       /*
        * Match all paths except:
        * - _next/static (static files)
        * - _next/image (image optimization)
        * - favicon.ico, sitemap.xml, robots.txt
        * - api routes
        */
       "/((?!_next/static|_next/image|favicon.ico|sitemap.xml|robots.txt|api).*)",
     ],
   };
   ```
   This matcher excludes /_next/* and /api/* at the framework level, so the middleware function only runs on page routes. The function then checks `PROTECTED_PREFIXES` to decide whether to enforce auth or pass through.
