    ```typescript
    import { z } from "zod";
    import { MUSICAL_KEYS } from "./song";

    export const filterSchema = z.object({
      bpmMin: z.coerce.number().int().min(1).max(500).optional(),
      bpmMax: z.coerce.number().int().min(1).max(500).optional(),
      key: z.enum(MUSICAL_KEYS).optional(),
      keySig: z.enum(["major", "minor"] as const).optional(),
      chord: z.string().max(100).optional(),
      lyric: z.string().max(500).optional(),
      tag: z.string().max(100).optional(),
    });

    export type FilterParams = z.infer<typeof filterSchema>;
    ```

    Key notes:
    - `z.coerce.number()` handles URL string-to-number conversion for BPM — no manual parseInt
    - `z.enum(MUSICAL_KEYS)` reuses the const from song.ts — do NOT duplicate the values
    - keySig is literally "keySig" (camelCase) — the URL param name matches this key
    - All fields are optional — no filter param is required
    - This file has no side effects — just exports the schema and type

  <action>
    Replace the GET handler in src/app/api/songs/route.ts to accept optional filter query params. The POST handler is unchanged — only GET changes.

    New GET handler:

    ```typescript
    import { NextResponse } from "next/server";
    import { db } from "@/db";
    import { songs, tags } from "@/db/schema";
    import { and, isNull, gte, lte, eq, ilike, exists, sql } from "drizzle-orm";
    import type { SQL } from "drizzle-orm";
    import { filterSchema } from "@/lib/validations/filter";
    import { songApiSchema } from "@/lib/validations/song";

    export async function GET(request: Request) {
      try {
        const { searchParams } = new URL(request.url);

        // Validate and coerce filter params — invalid values are ignored (safeParse)
        const parsed = filterSchema.safeParse(Object.fromEntries(searchParams));
        const f = parsed.success ? parsed.data : {};

        const conditions: SQL[] = [isNull(songs.deletedAt)];

        if (f.bpmMin !== undefined) conditions.push(gte(songs.bpm, f.bpmMin));
        if (f.bpmMax !== undefined) conditions.push(lte(songs.bpm, f.bpmMax));
        if (f.key) conditions.push(eq(songs.musicalKey, f.key as any));
        if (f.keySig) conditions.push(eq(songs.keySignature, f.keySig as any));

        if (f.chord) {
          conditions.push(
            sql`${songs.chordProgressions}::text ILIKE ${"%" + f.chord + "%"}`
          );
        }

        if (f.lyric) {
          // Use websearch_to_tsquery — NOT to_tsquery. to_tsquery fails on raw user input
          // e.g. "love song" would throw: syntax error in tsquery: "love song"
          conditions.push(
            sql`${songs.lyricsSearch} @@ websearch_to_tsquery('english', ${f.lyric})`
          );
        }

        if (f.tag) {
          // Use EXISTS subquery — NOT INNER JOIN. JOIN would duplicate rows when
          // a song has multiple tags that both match (e.g. tag="a" matches "ballad" AND "anthem")
          const tagSq = db
            .select({ id: sql<number>`1` })
            .from(tags)
            .where(and(eq(tags.songId, songs.id), ilike(tags.name, f.tag)));
          conditions.push(exists(tagSq));
        }

        const result = await db.query.songs.findMany({
          where: and(...conditions),
          with: { tags: true },
          orderBy: (songs, { desc }) => [desc(songs.createdAt)],
        });

        return NextResponse.json(result);
      } catch (err) {
        console.error("GET /api/songs error:", err);
        return NextResponse.json(
          { error: "Failed to fetch songs" },
          { status: 500 }
        );
      }
    }
    ```

    Important implementation notes:
    - The `and(...conditions)` call with `isNull(songs.deletedAt)` always present ensures soft deletes are always excluded
    - `f.bpmMin !== undefined` (not `if (f.bpmMin)`) because bpmMin=0 would be falsy — but Zod min(1) prevents 0 so `if (f.bpmMin)` is also safe. Use `!== undefined` for clarity
    - The BPM check `f.bpmMin !== undefined` and `f.bpmMax !== undefined` are independent — both, one, or neither can be present
    - Use `as any` for enum casts — Drizzle infers these as string literals, casting avoids complex type gymnastics
    - Do NOT use `db.select()` for the main query — `db.query.songs.findMany` with `with: { tags: true }` is the established pattern (same as before). The EXISTS subquery uses `db.select()` as a sub-expression only, which is correct
    - The POST handler is NOT changed — copy it verbatim from the existing file
